#Write a Python function to sum all the numbers in a list.
def add(num):
    result = 0
    for x in num:
        result += x
    return result
lst = []
# number of elements as input
n = int(input("Enter number of elements : "))
for i in range(0, n):
    ele = int(input())
    lst.append(ele) 
print("Sum of elements in list:",add(lst))
